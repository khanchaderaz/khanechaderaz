    <script>
        // نمایش تاریخ امروز به فارسی
        const dateEl = document.getElementById("todayDate");
        const months = ["فروردین","اردیبهشت","خرداد","تیر","مرداد","شهریور","مهر","آبان","آذر","دی","بهمن","اسفند"];
        const today = new Date();
        const persianDate = today.getDate() + " " + months[today.getMonth()] + " " + today.getFullYear();
        dateEl.textContent = "تاریخ امروز: " + persianDate;

        async function generatePDF() {
            const { jsPDF } = window.jspdf;
            const doc = new jsPDF();

            const day = document.getElementById("day").value;
            const traffic = document.getElementById("traffic").value;

            let color = "#2e7d32"; // پیش‌فرض سبز
            if(traffic === "متوسط") color = "#f1c40f";
            if(traffic === "زیاد") color = "#c0392b";

            doc.setFontSize(16);
            doc.text("گزارش ترافیک طرقبه و شاندیز", 105, 20, {align: "center"});

            doc.setFontSize(14);
            doc.text("تاریخ امروز: " + persianDate, 20, 40);
            doc.text("روز: " + day, 20, 50);
            
            doc.setTextColor(color);
            doc.text("ترافیک: " + traffic, 20, 60);
            doc.setTextColor(0,0,0); // بازگرداندن رنگ متن سیاه

            doc.save("Traffic_Report.pdf");
        }
    </script>
